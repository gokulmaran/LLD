⚙️ 1. System Calls (open, read, write, close)

Q1. What is a System Call? How is it different from a library function?
A: A system call is a controlled entry point into the kernel — it switches CPU from user mode to kernel mode to perform privileged operations.

Q2. Explain open(), read(), write(), close() with a complete example.
A:
cpp#include <fcntl.h>
#include <unistd.h>
#include <iostream>
using namespace std;

int main() {
    // open() — returns file descriptor
    // Flags: O_RDONLY, O_WRONLY, O_RDWR, O_CREAT, O_TRUNC, O_APPEND
    int fd = open("test.txt", O_CREAT | O_RDWR | O_TRUNC, 0644);
    if (fd < 0) {
        perror("open failed");
        return 1;
    }
    cout << "File descriptor: " << fd << "\n";   // e.g. 3

    // write() — returns bytes written, -1 on error
    const char* msg = "Hello, Linux!\n";
    ssize_t written = write(fd, msg, 14);
    cout << "Bytes written: " << written << "\n";

    // lseek() — rewind to beginning
    lseek(fd, 0, SEEK_SET);

    // read() — returns bytes read, 0 = EOF, -1 = error
    char buf[64] = {};
    ssize_t bytes = read(fd, buf, sizeof(buf));
    cout << "Read " << bytes << " bytes: " << buf;

    // close() — releases file descriptor
    close(fd);
    return 0;
}
// Output:
// File descriptor: 3
// Bytes written: 14
// Read 14 bytes: Hello, Linux!
File descriptor table:
Per-process FD table:
FD 0 ──► stdin   (keyboard)
FD 1 ──► stdout  (terminal)
FD 2 ──► stderr  (terminal)
FD 3 ──► test.txt  ← our open() call

Q3. What is the difference between open() flags O_TRUNC, O_APPEND, O_CREAT?
A:
cpp// O_CREAT  — create file if doesn't exist (needs mode arg)
int fd = open("a.txt", O_WRONLY | O_CREAT, 0644);

// O_TRUNC  — truncate existing file to 0 bytes on open
int fd = open("a.txt", O_WRONLY | O_TRUNC);

// O_APPEND — every write goes to end of file (atomic)
int fd = open("log.txt", O_WRONLY | O_APPEND);

// O_NONBLOCK — don't block on read/write (for pipes, sockets)
int fd = open("/dev/tty", O_RDWR | O_NONBLOCK);

// O_SYNC — every write flushed to disk before returning
int fd = open("safe.txt", O_WRONLY | O_SYNC);

Q4. What is errno? How do you use perror() and strerror()?
A: errno is a global integer set by system calls on failure. perror() and strerror() convert it to a human-readable message.
cpp#include <cerrno>
#include <cstring>
#include <fcntl.h>

int main() {
    int fd = open("/root/secret", O_RDONLY);
    if (fd < 0) {
        // Method 1: perror — prints to stderr automatically
        perror("open");                          // open: Permission denied

        // Method 2: strerror — returns string
        cout << "Error " << errno << ": "
             << strerror(errno) << "\n";         // Error 13: Permission denied

        // Common errno values:
        // ENOENT  = 2  — No such file or directory
        // EACCES  = 13 — Permission denied
        // EBADF   = 9  — Bad file descriptor
        // ENOMEM  = 12 — Out of memory
        // EINTR   = 4  — Interrupted by signal
    }
    return 0;
}

🔄 2. File Descriptors & I/O

Q5. What is dup() and dup2()? Give a real use case.
A: dup() creates a copy of a file descriptor. dup2() copies to a specific fd number — used to redirect stdin/stdout.
cpp#include <fcntl.h>
#include <unistd.h>

int main() {
    // Redirect stdout to a file
    int fd = open("output.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);

    // dup2(oldfd, newfd) — newfd now points to same file as oldfd
    dup2(fd, STDOUT_FILENO);   // fd 1 now points to output.txt
    close(fd);                  // original fd no longer needed

    // This now writes to output.txt, not terminal!
    write(STDOUT_FILENO, "This goes to file!\n", 19);

    return 0;
}
// output.txt contains: This goes to file!
Shell pipe implementation using dup2:
cpp// How shell implements: ls | grep txt
int pipefd[2];
pipe(pipefd);

if (fork() == 0) {          // child runs ls
    dup2(pipefd[1], STDOUT_FILENO);   // stdout → pipe write
    close(pipefd[0]);
    execlp("ls", "ls", nullptr);
} else {                    // parent runs grep
    dup2(pipefd[0], STDIN_FILENO);    // stdin ← pipe read
    close(pipefd[1]);
    execlp("grep", "grep", "txt", nullptr);
}

Q7. What is select() / poll() / epoll()? When do you use them?
A: These are I/O multiplexing mechanisms — monitor multiple file descriptors simultaneously without blocking.
Problem: Server needs to handle 1000 clients
         Can't do blocking read() on each!

Solution: I/O multiplexing — "tell me when any fd is ready"
cpp// select() — watch multiple fds (limited to FD_SETSIZE=1024)
fd_set readfds;
FD_ZERO(&readfds);
FD_SET(fd1, &readfds);
FD_SET(fd2, &readfds);

struct timeval tv = {5, 0};    // 5 second timeout
int ready = select(max_fd+1, &readfds, nullptr, nullptr, &tv);
if (FD_ISSET(fd1, &readfds)) { /* fd1 has data */ }

// poll() — no FD limit, cleaner API
struct pollfd fds[2];
fds[0] = {fd1, POLLIN, 0};   // watch fd1 for read
fds[1] = {fd2, POLLIN, 0};
int ready = poll(fds, 2, 5000);  // 5s timeout
if (fds[0].revents & POLLIN) { /* fd1 ready */ }

// epoll() — Linux only, O(1), scales to millions of fds
int epfd = epoll_create1(0);
struct epoll_event ev = {EPOLLIN, {.fd = fd1}};
epoll_ctl(epfd, EPOLL_CTL_ADD, fd1, &ev);

struct epoll_event events[10];
int n = epoll_wait(epfd, events, 10, 5000);
for (int i = 0; i < n; i++) {
    // events[i].data.fd is ready
}

🗺️ 3. Memory-Mapped Files (mmap)

Q8. What is mmap()? What are its uses?
A: mmap() maps a file or device into the process's virtual address space — access file like an array, no explicit read/write needed.
Without mmap:            With mmap:
read() → copy to buf     file pages mapped directly into VA space
buf[i]                   ptr[i]  ← same speed as memory access!

Physical RAM:
┌──────────────────────────────────┐
│  Page cache  │  Process pages   │
│  (file data) │  (mapped here)   │
└──────────────────────────────────┘
         ↑ Same physical pages — zero copy!
cpp#include <sys/mman.h>
#include <fcntl.h>
#include <unistd.h>
#include <iostream>
#include <cstring>
using namespace std;

int main() {
    int fd = open("data.bin", O_RDWR | O_CREAT, 0644);
    ftruncate(fd, 4096);       // set file size to 4KB

    // mmap(addr, length, prot, flags, fd, offset)
    char* ptr = (char*)mmap(
        nullptr,               // kernel chooses address
        4096,                  // map 4096 bytes
        PROT_READ | PROT_WRITE,// readable + writable
        MAP_SHARED,            // changes reflected in file
        fd,
        0                      // offset in file
    );

    if (ptr == MAP_FAILED) { perror("mmap"); return 1; }

    // Access file like memory!
    strcpy(ptr, "Hello mmap world!");
    cout << "Read back: " << ptr << "\n";

    // Flush changes to disk
    msync(ptr, 4096, MS_SYNC);

    munmap(ptr, 4096);        // unmap
    close(fd);
    return 0;
}

Q9. What is the difference between MAP_SHARED and MAP_PRIVATE?
A:
cpp// MAP_SHARED — writes visible to other processes + written to file
char* p1 = mmap(0, size, PROT_WRITE, MAP_SHARED, fd, 0);
p1[0] = 'X';    // ← file on disk updated, other processes see 'X'

// MAP_PRIVATE — copy-on-write, changes NOT reflected to file
char* p2 = mmap(0, size, PROT_WRITE, MAP_PRIVATE, fd, 0);
p2[0] = 'X';    // ← private copy created, file unchanged

// MAP_ANONYMOUS — no file backing (like malloc but page-aligned)
char* p3 = mmap(0, size, PROT_READ|PROT_WRITE,
                MAP_PRIVATE|MAP_ANONYMOUS, -1, 0);
// Use for large allocations or shared memory between fork'd processes

Q10. What is the difference between blocking and non-blocking sockets?
A:
cpp// Blocking socket (default)
// read() waits until data arrives — thread is blocked!

// Now read() returns immediately
ssize_t n = read(fd, buf, 1024);
if (n < 0 && errno == EAGAIN) {
    // No data available right now — try later
    // Use select/poll/epoll to wait efficiently
}

Q11. What is the SO_REUSEADDR socket option? Why is it important?
A: After a server crashes/restarts, the port stays in TIME_WAIT state for ~60 seconds. SO_REUSEADDR lets you rebind immediately.
cppint server_fd = socket(AF_INET, SOCK_STREAM, 0);

// Set before bind() — always do this for servers!
int opt = 1;
setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR,
           &opt, sizeof(opt));

// Now bind works even if port is in TIME_WAIT
bind(server_fd, (sockaddr*)&addr, sizeof(addr));

🔔 5. Signals (Deep Dive)

Q12. What is signal masking? How do you block signals temporarily?
A: Signal masking lets you temporarily block signals during critical sections.
cpp#include <csignal>
#include <unistd.h>

int main() {
    sigset_t mask, old_mask;

    sigemptyset(&mask);
    sigaddset(&mask, SIGINT);    // add SIGINT to mask
    sigaddset(&mask, SIGTERM);   // add SIGTERM to mask

    // Block SIGINT and SIGTERM
    sigprocmask(SIG_BLOCK, &mask, &old_mask);

    // Critical section — signals are blocked here
    // (they are queued, not lost)
    cout << "In critical section — signals blocked\n";
    sleep(3);

    // Restore original mask — queued signals delivered now
    sigprocmask(SIG_SETMASK, &old_mask, nullptr);

    cout << "Signals unblocked\n";
    return 0;
}
=====================================================================
Q13. What is sigsuspend()? How do you safely wait for a signal?
A: sigsuspend() atomically unblocks signals and suspends the process — avoids race condition between unblocking and sleeping.
cpp// WRONG — race condition!
sigprocmask(SIG_UNBLOCK, &mask, nullptr);
pause();   // ← signal could arrive HERE before pause()!

// CORRECT — atomic unblock + sleep
sigset_t empty;
sigemptyset(&empty);
sigsuspend(&empty);   // atomically: unblock all + sleep until signal

📋 8. POSIX APIs

Q21. What are POSIX threads (pthreads)?
A: POSIX threads provide a standardized threading API across Unix-like systems.
cpp#include <pthread.h>
#include <iostream>
using namespace std;

struct Args { int id; int result; };

void* worker(void* arg) {
    Args* a = (Args*)arg;
    cout << "Thread " << a->id << " running\n";
    a->result = a->id * a->id;
    return nullptr;
}

int main() {
    pthread_t threads[4];
    Args args[4];

    // Create threads
    for (int i = 0; i < 4; i++) {
        args[i].id = i;
        pthread_create(&threads[i], nullptr, worker, &args[i]);
    }

    // Join threads
    for (int i = 0; i < 4; i++) {
        pthread_join(threads[i], nullptr);
        cout << "Result " << i << " = " << args[i].result << "\n";
    }

    // Mutex usage
    pthread_mutex_t mtx = PTHREAD_MUTEX_INITIALIZER;
    pthread_mutex_lock(&mtx);
    // critical section
    pthread_mutex_unlock(&mtx);
    pthread_mutex_destroy(&mtx);

    return 0;
}
// Compile: g++ -pthread prog.cpp
