Splitwise — Low-Level Design

System Design Reference · C++ Implementation

1. Requirements

Functional Core

  • Add users
  • Create groups
  • Add expenses (equal / exact / percentage)
  • Split expenses among users
  • Simplify debts (who owes whom)
  • Show balances

Non-Functional Quality

  • Scalable — handles many users/groups
  • Consistent calculations — no rounding bugs
  • Extensible — easy to add new split types

2. Core Entities & Responsibilities

User

  • id, name, email
  • tracks balances with other users

Expense

  • id, amount, paidBy
  • list of splits

Split (abstract)

  • user + amount
  • EqualSplit
  • ExactSplit
  • PercentSplit

Group

  • id, users
  • expenses

BalanceSheet

  • who owes whom
  • Map<User, Map<User, double>>

Splitwise (Service)

  • addExpense()
  • showBalances()
  • orchestrates everything

3. Class Diagram (Mental Model)

User
└── id, name

Expense
├── amount
├── paidBy
└── List<Split>

Split (abstract)
├── User user
└── double amount
↑
├── EqualSplit
├── ExactSplit
└── PercentSplit

Group
├── List<User>
└── List<Expense>

BalanceSheet
└── Map<User, Map<User, double>>

Splitwise
├── addExpense()
└── showBalances()
Key Design Decision: Use abstract Split class → extensibility. Keep balance logic centralized in BalanceSheet.

4. C++ Implementation

Base Classes

#include <bits/stdc++.h>
using namespace std;

class User {
public:
    string id;
    string name;

    User(string id, string name) : id(id), name(name) {}
};

Split Hierarchy

class Split {
public:
    User* user;
    double amount;

    Split(User* user) : user(user), amount(0) {}
    virtual ~Split() {}
};

class EqualSplit : public Split {
public:
    EqualSplit(User* user) : Split(user) {}
};

class ExactSplit : public Split {
public:
    ExactSplit(User* user, double amount) : Split(user) {
        this->amount = amount;
    }
};

class PercentSplit : public Split {
public:
    double percent;

    PercentSplit(User* user, double percent) : Split(user), percent(percent) {}
};

Expense

enum ExpenseType {
    EQUAL,
    EXACT,
    PERCENT
};

class Expense {
public:
    string id;
    double amount;
    User* paidBy;
    vector<Split*> splits;
    ExpenseType type;

    Expense(string id, double amount, User* paidBy,
            vector<Split*> splits, ExpenseType type)
        : id(id), amount(amount), paidBy(paidBy),
          splits(splits), type(type) {}
};

Balance Sheet

class BalanceSheet {
public:
    unordered_map<string, unordered_map<string, double>> balance;

    void updateBalance(User* paidBy, Split* split) {
        string payer = paidBy->id;
        string payee = split->user->id;

        if (payer == payee) return;

        balance[payee][payer] += split->amount;
        balance[payer][payee] -= split->amount;
    }

    void showBalances() {
        for (auto &p : balance) {
            for (auto &q : p.second) {
                if (q.second > 0) {
                    cout << p.first << " owes " << q.first
                         << " : " << q.second << endl;
                }
            }
        }
    }
};

Splitwise Service (Core Logic)

class Splitwise {
private:
    BalanceSheet balanceSheet;

public:
    void addExpense(string id, double amount, User* paidBy,
                    vector<Split*> splits, ExpenseType type) {

        int n = splits.size();

        if (type == EQUAL) {
            double splitAmount = amount / n;
            for (auto split : splits) {
                split->amount = splitAmount;
            }
        }
        else if (type == PERCENT) {
            for (auto split : splits) {
                PercentSplit* ps = dynamic_cast<PercentSplit*>(split);
                split->amount = (ps->percent * amount) / 100.0;
            }
        }
        // EXACT → already assigned

        // Validate
        double total = 0;
        for (auto split : splits) total += split->amount;

        if (abs(total - amount) > 0.001) {
            throw runtime_error("Invalid split");
        }

        // Update balances
        for (auto split : splits) {
            balanceSheet.updateBalance(paidBy, split);
        }
    }

    void showBalances() {
        balanceSheet.showBalances();
    }
};

Example Usage

int main() {
    User u1("u1", "A");
    User u2("u2", "B");
    User u3("u3", "C");

    Splitwise sw;

    vector<Split*> splits;
    splits.push_back(new EqualSplit(&u1));
    splits.push_back(new EqualSplit(&u2));
    splits.push_back(new EqualSplit(&u3));

    sw.addExpense("e1", 3000, &u1, splits, EQUAL);

    sw.showBalances();
}
